async function getTableCell() {
  await Word.run(async (context) => {
    let myTable = context.document.body.tables.getFirst();
    myTable.load("rowCount");
    await context.sync();
    for (var i = 0; i < myTable.rowCount; i++) {
      var lastCell = myTable.getCell(i, 3);
      var lastCellBody = lastCell.body;
      var myText = lastCellBody.load("text");
      await context.sync();
      if (lastCellBody.text == "Verde") {
        lastCell.shadingColor = "#00FF00";
      } else if (lastCellBody.text == "Giallo") {
        lastCell.shadingColor = "#FFFF00";
      } else if (lastCellBody.text == "Rosso") {
        lastCell.shadingColor = "#FF0000";
      }
    }
  });
}
 
/** Default helper for invoking an action and handling errors. */
async function tryCatch(callback) {
  try {
    await callback();
  } catch (error) {
    // Note: In a production add-in, you'd want to notify the user through your add-in's UI.
    console.error(error);
  }
}
